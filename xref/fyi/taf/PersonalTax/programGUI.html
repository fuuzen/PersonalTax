36       * GUI program main function
37       * @param args 参数无效,不需要
38       */
39      public static void main(String[] args) {
40          Calculator calculator = new Calculator();
41  
42          JFrame frame = new JFrame("个人所得税计算器");
43          frame.setSize(650, 300);
44          frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
45  
46          JPanel panel = new JPanel();    
47          frame.add(panel);
48          panel.setLayout(null);
49  
50          try {
51              String os = System.getProperty("os.name").toLowerCase();
52              Image icon;
53              if (os.contains("win")) {
54                  ImageIcon ico = new ImageIcon("images/icons/icon.ico");
55                  icon = ico.getImage();
56              } else if (os.contains("mac")) {
57                  ImageIcon icns = new ImageIcon("images/icons/icon.icns");
58                  icon = icns.getImage();
59              } else {
60                  InputStream inputStream16 = programGUI.class.getClassLoader().getResourceAsStream("images/icons/icon16.png");
61                  InputStream inputStream32 = programGUI.class.getClassLoader().getResourceAsStream("images/icons/icon32.png");
62                  InputStream inputStream64 = programGUI.class.getClassLoader().getResourceAsStream("images/icons/icon64.png");
63                  InputStream inputStream128 = programGUI.class.getClassLoader().getResourceAsStream("images/icons/icon128.png");
64                  InputStream inputStream256 = programGUI.class.getClassLoader().getResourceAsStream("images/icons/icon256.png");
65  
66                  if (inputStream16 == null || inputStream32 == null || inputStream64 == null || inputStream128 == null || inputStream256 == null) {
67                      throw new IOException("Icon not found!");
68                  }
69  
70                  BufferedImage icon16 = ImageIO.read(inputStream16);
71                  BufferedImage icon32 = ImageIO.read(inputStream32);
72                  BufferedImage icon64 = ImageIO.read(inputStream64);
73                  BufferedImage icon128 = ImageIO.read(inputStream128);
74                  BufferedImage icon256 = ImageIO.read(inputStream256);
75      
76                  icon = new BaseMultiResolutionImage(icon16, icon32, icon64, icon128, icon256);
77              }
78              frame.setIconImage(icon);
79  
80              InputStream inputStream = programGUI.class.getClassLoader().getResourceAsStream("images/taffy.png");
81              if (inputStream == null) {
82                  throw new IOException("Image not found!");
83              }
84              JLabel taffy = new JLabel(
85                  new ImageIcon(
86                      ImageIO.read(inputStream).getScaledInstance(250, 250, Image.SCALE_SMOOTH)
87                  )
88              );
89              taffy.setBounds(25, 5, 250, 250);
90              panel.add(taffy);
91          } catch (IOException e) {
92              e.printStackTrace();
93              JOptionPane.showMessageDialog(frame, "Failed to load image!", "Error", JOptionPane.ERROR_MESSAGE);
94          }
95  
96          JLabel inputLabel = new JLabel("您的收入: ");
97          inputLabel.setBounds(300, 20, 100, 25);
98          inputLabel.setHorizontalAlignment(JLabel.RIGHT);
99          panel.add(inputLabel);
100 
101         SpinnerNumberModel input = new SpinnerNumberModel(0, 0, Integer.MAX_VALUE, 1);
102         JSpinner spinner = new JSpinner(input);
103         spinner.setBounds(400, 20, 150, 25);
104         panel.add(spinner);
105 
106         JButton calButton = new JButton("计算");
107         calButton.setBounds(400, 50, 100, 25);
108         panel.add(calButton);
109 
110         JLabel texLabelText = new JLabel("您的个人所得税为: ");
111         texLabelText.setBounds(350, 80, 200, 25);
112         texLabelText.setHorizontalAlignment(JLabel.CENTER);
113         panel.add(texLabelText);
114 
115         String[] columnNames = {"税级", "起征点", "税率", "您应缴税"};
116         DefaultTableModel model = new DefaultTableModel(columnNames, 0) {
117             @Override
118             public boolean isCellEditable(int row, int column) {
119                 // 不可编辑
120                 return column != 0 && column != 3 && row != 0 && row != 6;
121             }
122 
123             @Override
124             public void setValueAt(Object value, int row, int column) {
125                 super.setValueAt(value, row, column);
126                 if (column == 1) { // 修改起征点
127                     calculator.adjustTaxThreshold(row - 1, Double.parseDouble(value.toString()));
128                 }
129                 if (column == 2) { // 修改税率
130                     calculator.adjustTaxRate(row - 1, Double.parseDouble(value.toString()));
131                 }
132             }
133         };
134         model.addRow(new Object[]{"税级", "起征点", "税率", "您应缴税"});
135         for(int i = 0; i < calculator.getTaxLevelCount(); ++i) {
136             TaxLevel taxLevel = calculator.getTaxLevel(i);
137             model.addRow(new Object[]{i + 1, taxLevel.getThreshold(), taxLevel.getRate(), 0});
138         }
139         model.addRow(new Object[]{"总共", null, null, 0});
140         JTable table = new JTable(model);
141         table.setBounds(300, 120, 300, 120);
142         panel.add(table);
143 
144         calButton.addActionListener(e -> {
145             ArrayList<Double> res = calculator.calculate(input.getNumber().doubleValue());
146             for(int i = 0; i < 6; ++i) {
147                 model.setValueAt(String.format("%,.2f", res.get(i)), i + 1, 3);
148             }
149         });
150 
151         frame.setVisible(true);
152     }
